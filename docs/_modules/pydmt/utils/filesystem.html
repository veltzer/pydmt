    for file in files:
        if only_if_exist:
            if os.path.isfile(file):
                os.unlink(file)

            os.unlink(file)




[docs]def files_under_folder(folder: str, suffix: Union[str, None] = None) -> List[str]:
    file_list = []
    for dir_path, _, filenames in os.walk(folder):
        for filename in filenames:
            if suffix is None or filename.endswith(suffix):
                file_list.append(os.path.join(dir_path, filename))
    return file_list




[docs]def files_under_folders(folders: List[str]) -> List[str]:
    file_list = []
    for folder in folders:
        file_list.extend(files_under_folder(folder))
    return file_list
