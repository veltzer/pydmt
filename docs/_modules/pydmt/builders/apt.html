        os.environ['DEBIAN_FRONTEND'] = 'noninteractive'
        args = []
        if ConfigSudo.sudo:
            args.append("sudo")
        args.extend([
            'apt-get',
        ])
        if ConfigApt.apt_quiet:
            args.append("-q=2")
        args.extend([
            '--yes',
            'update',
        ])
